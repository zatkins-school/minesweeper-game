 12 /**
 13 	* this takes an x value, an array, and the current number of elements in that array.
 14 	* It checks (for the number of elements currently in the array) whether or not the value
 15 	* is already in the array. If it is then it returns false, if it is not it returns true.
 16 	* @param {number} value - Value that is interpreted as a mine
 17 	* @param {array} arr - the grid of the game
 18 	* @param {number} currentLength - size of the grid
 19 	* @return {boolean} unique - if value is in array, return true
 20 	* @pre: need a valid value, array, and array length
 21 	* @post: checks to see if value is in array
 22 */
 23 function unique_func(value, arr, currentLength){
 24 	let unique = true;
 25 	for(let i=0; i<currentLength; i++){
 26 		if(value == arr[i]){
 27 			unique = false;
 28 		}
 29 	}
 30 	return unique;
 31 }

 33 /**
 34 	* this first creates an array of unique random numbers less than the floor of rows*cols
 35 	* when isFull=true the array of mine locations is full, its time to put them in!
 36 	* The triple nested for-loop places all the mines in place. We chose to use a "9"
 37 	* to represent a 9 because all spaces around it =1 plus it is a mine so 8+1=9
 38 	* @param {number} number - number of mines
 39 	* @param {number} rows - number of rows
 40 	* @param {number} cols - number of cols
 41 	* @param {array} multi_array - the grid of the game
 42 	* @pre: need valid number of mines, rows, cols, and a grid
 43 	* @post: populates mines on the grid
 44  */
 45 function mine_population(number, rows, cols, multi_array){
 46 	let mine_locations = new Array(number); /**creates empty array the size of the number of mines the user picked*/
 47 	let isfull = false; /**this will keep track of when the mine_locations array is full (currently empty)*/
 48 	let mineCount=1; /**this needs to start at 1 so that unique_func can run the first time*/
 49 	let i=0;
 50 	while(!isfull){ /**only run until the mine_locations array is full*/
 51 		let x=Math.floor(Math.random() * (rows*cols)); /**this creates a random int between 0 and rows*cols then takes the floor*/
 52 
 53 		if(unique_func(x, mine_locations, mineCount)){ /**if unique, run below code*/
 54 			mine_locations[i] = x; /**place this x value in mine_locations*/
 55 			mineCount++; /**increment the number of mines that have been placed*/
 56 			i++;
 57 			if(mineCount-1 == number){ /**-1 accounts for starting at 1*/
 58 				isfull = true; /**array is full, you can break out and put them in now*/
 59 			}
 60 		}
 61 	}
 62 	for(let i=0; i<number; i++){ /**for the number of mines they would like to place*/
 63 		let y = mine_locations[i] % (cols);
 64 		let x = (mine_locations[i] - y) / (cols);
 65 		multi_array[y][x].mine=true;
 66 	}
 67 }

 69 /**
 70 	* This function shuffles the mines among the unrevealed spaces left on the board.
 71 	* @param {array} multi_array - the grid of the game
 72 	* @pre: need valid number of mines, rows, cols, and a grid
 73 	* @post: populates mines on the grid
 74  */
 75 function mine_shuffle(multi_array){
 76 	let numMinesLeft = 0;
 77 	let totalMines = 0;
 78 	let unrevealedSpacesArray = [];
 79 
 80 	//find the number of mines left unrevealed
 81 	for(let i = 0; i < multi_array.length; i++){
 82 		for(let j = 0 ; j < multi_array[i].length; j++){
 83 			if(!multi_array[i][j].revealed && !multi_array[i][j].flagged){
 84 				//create an array of unrevealed spaces
 85 				unrevealedSpacesArray.push(multi_array[i][j]);
 86 
 87 				if(multi_array[i][j].mine){
 88 					//add to our count of mines
 89 					numMinesLeft++;
 90 					//remove the mine from this space
 91 					multi_array[i][j].mine = false;
 92 				}
 93 			}
 94 			if(multi_array[i][j].mine){
 95 				totalMines++;
 96 			}
 97 		}
 98 	}
 99 	
100 	let newMineArray = [];
101 	//generate random numbers to designate where the mines will be shuffled to
102 	let shufflingMines = true;
103 	while(shufflingMines){
104 		//Check if we've found a location for every mine
105 		if(newMineArray.length === numMinesLeft){
106 			shufflingMines = false;
107 		} else {
108 			//If not, find a new location for a new mine
109 			let spaceIndex = Math.floor(Math.random()*unrevealedSpacesArray.length);
110 			if(unique_func(spaceIndex, newMineArray, newMineArray.length)){
111 				newMineArray.push(spaceIndex);
112 			}
113 		}
114 	}
115 
116 	//loop through those indecies and mark them as mines
117 	for(let i = 0; i < newMineArray.length; i++){
118 		unrevealedSpacesArray[newMineArray[i]].mine = true;
119 	}
120 
121 	//Regenerate the field to move the numbers as well
122 	generate_playing_field(totalMines, rows, cols, multi_array);
123 }

125 /**
126 	* checks to see if a given index of a 2d array is within that array's bounds
127 	* @param {number} a - bound variable for the rows of grid
128 	* @param {number} b - bound variable for the of grid
129 	* @param {number} cols - number of cols
130 	* @param {number} rows - number of rows
131 	* @return {boolean} - if in bounds, return true
132 	* @pre: need valid rows, cols, and bounds
133 	* @post: nothing
134 */
135 function is_within_bounds(a, b, cols, rows){
136 	if((a >= 0) && (b >= 0) && (a < cols) && (b < rows)){
137 		return true;
138 	}
139 	else{
140 		return false;
141 	}
142 }

146 /** Adds adjacent numbers to the playing grid
147 	* @param {number} number_of_mines - the number of mines
148 	* @param {number} rows - number of rows
149 	* @param {number} cols - number of cols
150 	* @param {array} multi_array - the grid of the game
151 	* @pre: need valid mines, rows, cols, and grid
152 	* @post: places numbers on the grid that are adjacent to mines
153 	*/
154 function generate_playing_field(number_of_mines, rows, cols, multi_array){
155 	/**these two for loops traverse through every index of the array */
156 	for (let c = 0; c < cols; c++) {
157 	  for (let r = 0; r < rows; r++) {
158 			let count=0;
159 			if(multi_array[c][r].mine != true){
160 				/** these two for loops check all surrounding indices of the current index in the array */
161 				for(let a = -1; a <= 1; a++){
162 					for(let b = -1; b <= 1; b++){
163 						let column_test = c + a;
164 						let row_test = r + b;
165 						if(is_within_bounds((column_test),(row_test),cols,rows)){
166 							if(multi_array[c+a][r+b].mine == true){
167 								count = count +1;
168 							}
169 						}
170 					}
171 				}
172 			}
173 			/** assigns the number of adjacent mines to the current index*/
174 			multi_array[c][r].count = count;
175 	  }
176 	}
177 }

179 /**
180  * Reveals boxes around grid [y,x] that are spaces, and for each space has a recursive call to reveal its spaces
181  * @param   {number} y column coordinate of the space
182  * @param   {number} x row coordinate of the space
183  * @param   {number} cols number of columns on the grid
184  * @param   {number} rows number of rows on the grid
185  * @param   {Array} multi_array grid
186  * @pre: need valid cols, rows, multi_array, and grid coordinates
187  * @post: recursively reveals spaces
188  */
189 function reveal_spaces(y, x, cols, rows, multi_array){
190 	/** Nested for loop to access the boxes around grid[y][x] */
191 	for(let a = y-1; a <= y+1; a++){
192 		for(let b = x-1; b <= x+1; b++){
193 			if(is_within_bounds(a,b,cols,rows)){
194 				/** click the boxes that have count=0, haven't been cliked and are not a mine or flag */
195 				if(multi_array[a][b].count == 0 && grid[a][b].revealed==false && multi_array[a][b].mine == false && multi_array[a][b].flagged == false){
196 					multi_array[a][b].revealed = true;
197 					multi_array[a][b].hasBeenRevealed = true;
198 					/** recursive call to reveal spaces of the new count=0 box */
199 					reveal_spaces(a,b,cols,rows,multi_array);
200 				}
201 
202 				/** reach a number boundarie, if it is not a bomb or flag click the box */
203 				else if (multi_array[a][b].mine==false && multi_array[a][b].count>0 && multi_array[a][b].flagged == false){
204 					multi_array[a][b].revealed = true;
205 					multi_array[a][b].hasBeenRevealed = true;
206 				}
207 			}
208 		}
209 	}

213 /** Let's user know that the game is finished and resets the game
214 	* @param {number} cols - number of columns
215 	* @param {number} rows - number of rows
216 	* @param {array} multi_array - grid of the game
217 	* @param {number} flags - number of flags on the grid
218 	* @return {boolean} - true if the user wins
219 	* @pre: need valid rows, cols, grid, and flags
220 	* @post: allows another method to indicate if the user won
221 	*/
222 function win(cols, rows, multi_array, flags){
223 	let count = 0;
224 	for(let i = 0; i < cols; i++){
225 		for(let j = 0; j < rows; j++){
226 			if(multi_array[i][j].flagged == true && multi_array[i][j].mine == true){
227 				count += 1;
228 			}
229 		}
230 	}
231 
232 	if(count == flags){
233 		return true;
234 	}
235 	else{
236 		return false;
237 	}
