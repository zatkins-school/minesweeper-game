1 /**
  2  * @fileOverview Implementation of the music functions
  3 */
  4 
  5 /** ------------Global variable ------------ */
  6 let state = null;
  7 
  8 /** @function
  9  * When state is null, creates an audio tag for the music and starts the music.
 10  * when state is true it does nothing, and when state is false it unpauses the music.
 11  * @pre Music file must exist
 12  * @post Audio tag in the HTML
 13 */
 14 function audioCreator()
 15 {
 16 	if(state === null)
 17 	{
 18 		this.music = document.createElement("audio");
 19 		this.music.src = "../minesweeper-game/sound/bensound-highoctane.mp3";
 20 		this.music.setAttribute("preload", "auto");
 21 		this.music.setAttribute("controls", "none");
 22 		this.music.setAttribute("loop", "true");
 23 		this.music.setAttribute("id", "intenseAudio");
 24 		this.music.style.display = "none";
 25 		document.body.appendChild(this.music);
 26 		this.music.play();
 27 		state = true;
 28 	}
 29 	else
 30 	{
 31 		document.getElementById("intenseAudio").play();
 32 		state = true;
 33 	}
 34 }
 35 
 36 /** @function
 37 	* When state is null or false it does nothing. When state is true it pauses the music and
 38 	* sets the state to false.
 39 	* @pre Audio tag must exist if state is true
 40 	* @post Music is paused
 41 */
 42 function audioStopper()
 43 {
 44 		document.getElementById("intenseAudio").pause();
 45 		state = false;
 46 }
