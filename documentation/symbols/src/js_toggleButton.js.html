  7 /**
  8  * toggles button text and checkbox state between
  9  * normal and intense mode
 10  * 
 11  * toggles intense mode
 12  * @pre the game must be initialized
 13  * @post switches between intense mode and normal mode
 14  * @return none
 15  */
 16 function toggleIntenseMode() {
 17     let button = document.getElementById("intenseMode");
 18     let checkBox = document.getElementById("intenseModeInput");
 19     let titleText = document.getElementById('bigText');
 20     if (button.innerText==="Normal Mode") {
 21         titleText.innerHTML = "SUPER CRAZY LUDICROUS INTENSE MINESWEEPER WOW";
 22         shakeIt();
 23         button.innerText = "INTENSE MODE!!";
 24         button.style.backgroundColor = "red";
 25         checkBox.checked = true;
 26         audioCreator();
 27         document.body.style.backgroundImage = "url('../minesweeper-game/css/intenseModeHellfire.jpg')";
 28         document.getElementById("introtext").hidden = true;
 29         document.getElementById("intenseRules").hidden = false;
 30         let ptags = document.getElementById("settings").children;
 31         for (let elem of ptags){
 32             elem.classList.add("intense");
 33         }
 34         let intenseTexts = document.getElementsByClassName("isIntense");
 35         for (let elem of intenseTexts) {
 36             elem.classList.add("intense");
 37         }
 38     } else {
 39         titleText.innerHTML = "MINESWEEPER";
 40         stopAnimation();
 41         button.innerText = "Normal Mode";
 42         button.style.backgroundColor = "#ffce00";
 43         checkBox.checked = false;
 44         audioStopper();
 45         document.body.style.backgroundImage = "url('../minesweeper-game/css/background.jpg')";
 46         document.getElementById("introtext").hidden = false;
 47         document.getElementById("intenseRules").hidden = true;
 48         let ptags = document.getElementById("settings").children;
 49         for (let elem of ptags){
 50             elem.classList.remove("intense");
 51         }
 52         let intenseTexts = document.getElementsByClassName("isIntense");
 53         for (let elem of intenseTexts) {
 54             elem.classList.remove("intense");
 55         }
 56     }
 57 }

 59 /**

 61  * 
 62  * toggles cheats enabled and ghost-reveals all spaces
 63  * 
 64  * @pre the game must be initialized
 65  * @post enables/disables cheats
 66  * @return none
 67  */
 68 function toggleCheats(){
 69     let button = document.getElementById("cheatMode");
 70     let checkBox = document.getElementById("cheatModeInput");
 71     if(cheating === true){
 72         cheating = false;
 73         button.innerText = "Cheats Off";
 74         button.style.backgroundColor = "#ffce00";
 75         checkBox.checked = false;
 76     }
 77     else{
 78         cheating = true;
 79         button.innerText = "Cheats On";
 80         button.style.backgroundColor = "red";
 81         checkBox.checked = true;
 82     }
 83     for(x = 0; x < cols; x++){
 84         for(y = 0; y < rows; y++){
 85             if(cheating === true){
 86                 grid[x][y].revealed = true;
 87             }
 88             else{
 89                 if(grid[x][y].hasBeenRevealed === false){
 90                     grid[x][y].revealed = false;
 91                 }
 92             }
 93         }
 94     }
 95     redraw();
 96 }
