 26 /** Creates a canvas with a 2D array according to the input
 27     * @pre need a valid number of rows and columns to build the board
 28     * @post the board has been created and filled with mines, the user can now play the game!
 29     */
 30 function setup() {
 31 
 32     /** Resets cheating button */
 33     let button = document.getElementById("cheatMode");
 34     let checkBox = document.getElementById("cheatModeInput");
 35     button.innerText = "Cheats Off";
 36     button.style.backgroundColor = "#ffce00";
 37     checkBox.checked = false;
 38     /** Reset globals */
 39     reset();
 40     loop();
 41     /** Gets the dimensions from the user */
 42     rows = floor(document.getElementById("input1").value);
 43     cols = floor(document.getElementById("input2").value);
 44     mines = floor(document.getElementById("input3").value);
 45     flags = mines;
 46 
 47     /** Boundaries for the grid */
 48     if (rows<2 || rows>30){
 49         sizeError();
 50         return;
 51     }
 52     if (cols < 2 || cols > 30){
 53         sizeError();
 54         return;
 55     }
 56     if(mines<=0){
 57         minesError();
 58         return;
 59     }
 60     if(mines>=rows*cols){
 61         minesError();
 62         return;
 63     }
 64     width = max(90 - 2.5*max(rows, cols), 30);
 65     /** Clear errors */
 66     document.getElementById("minesBoundsError").hidden = true;
 67     document.getElementById("sizeBoundsError").hidden = true;
 68     /** Creates a canvas that holds the amount of cols and rows given according to the set width*/
 69     let canvas = createCanvas(cols*width +1, rows*width +1);
 70     canvas.parent('canvas-holder');
 71 
 72     /** Creates a 2D Array with the cols and rows given*/
 73     grid = create2DArray(cols, rows);
 74     for (let c = 0; c < cols; c++) {
 75       for (let r = 0; r < rows; r++) {
 76         grid[c][r] = new Box(c*width, r*width, width);
 77       }
 78     }
 79 
 80     /** Populates the grid with the amount of mines given */
 81     mine_population(mines, rows, cols, grid);
 82 
 83     /** Populates the count of each box in the grid */
 84     generate_playing_field(mines, rows, cols, grid);
 85     
 86 }

118 /**
119  * calculates the maximum number of mines based on current input
120  *
121  * @returns {number} rows*cols-1, max number of mines
122  */
123 function getMineMax() {
124     return  document.getElementById("input1").value * document.getElementById("input2").value - 1;
125 }
126 
127 /** Takes in the provided board size and mine count from the user and validates that against the set boundaries
128   * @pre assumes that the inputs contain numbers
129   * @post shows error text if aplicable, and sets the max number of mines possible within the given size
130   */
131 function checkInputs() {
132     let t_rows = document.getElementById("input1").value;
133     let t_cols = document.getElementById("input2").value;
134     let mineElem = document.getElementById("input3");
135     let t_mineMax;
136     if (t_rows<2 || t_rows>30 || t_cols<2 || t_cols>30) {
137         sizeError();
138         t_mineMax = 1;
139     } else {
140         document.getElementById("sizeBoundsError").hidden = true;
141         t_mineMax = t_rows*t_cols - 1;
142     }
143     mineElem.max = t_mineMax;
144 }

163 /** Draws the canvas on the site by calling the show function on each box
164     * @pre there has been a 2d array built, but it has nothing inside it
165     * @post the array is now filled with mines or number of adjacent mines
166     * @return none
167     */
168 function draw() {
169     for (let c = 0; c < cols; c++) {
170         for (let r = 0; r < rows; r++) {
171         grid[c][r].show();
172         }
173     }
174     document.getElementById("flagsLeft").innerHTML = ("Remaining flags: " + flags);
175     if (isGameover) {
176         if (isWin) {
177             window.alert("You win!");
178         }
179         else {
180             window.alert("You lose.");
181         }
182         noLoop();
183         reset();
184     }
185 }

189 /**
190  * Creates a 2D Array according to the cols and rows given
191  * @param   {number} cols number of columns for the 2D array
192  * @param   {number} rows number of rows for the 2D array
193  * @return  {array} 2D array with the the numbers of cols and rows
194  */
195 function create2DArray(cols,rows){
196     let grid = new Array(cols);
197     for (let c=0; c<cols; c++){
198         grid[c] = new Array(rows);
199     }
200     return grid;
201 }

204 /** Places a flags where the mouse is located when f is released
205   * @pre An array has been created and filled with box objects, some of which contain mines
206   * @post Once the f key has been released a flag will be set wherever the cursor hovered when f is pressed
207   *       (decrementing flagCount by 1) if this is repeated, the flag will be removed and flagCount will increase by 1
208   * @return none
209 */
210 function flag(x,y) {
211     if (isGameover) return;
212     try {
213         grid[x][y].flagged;
214     } catch (e) {
215         return;
216     }
217     if (grid[x][y].revealed) {
218         return;
219     }
220     if (grid[x][y].flagged){
221         grid[x][y].flagged=false;
222         flags = flags + 1;
223     }
224     else if (flags > 0){
225         grid[x][y].flagged=true;
226         flags = flags - 1;
227     }
228     if (win(cols, rows, grid, mines)) {
229         grid[x][y].show();
230         isGameover = true;
231         isWin = true;
232     }
233 }

236 /** Reveals a space clicked on by the user
237   * @pre The game board has been initialized, the game has begun
238   * @post triggers winning, losing, or the waterfall reveal if necessary, triggers mine shuffle if we are in intense mode
239   * @return none
240 */
241 function reveal(x,y) {
242  
243     
244     if (isGameover) return;
245     if (x<0 || x>cols-1 || y<0 || y>rows-1) return;
246     if (grid[x][y].flagged || grid[x][y].revealed) return;
247     grid[x][y].revealed=true;
248     grid[x][y].hasBeenRevealed = true;
249     /** Calls lose function */
250     if (grid[x][y].mine) {
251         grid[x][y].show();
252         isGameover = true;
253         return;
254     }
255    
256     /** Generates spaces if the box is an space and not a mine*/
257     else if (grid[x][y].count==0) {
258         reveal_spaces(x,y,cols,rows,grid);
259     }
260     /** Calls win function */
261     if (win(cols, rows, grid, mines)) {
262         grid[x][y].show();
263         isGameover = true;
264         isWin = true;
265     }
266 
267     //I'm sorry Zach no ur not
268     if(isIntense()){
269         mine_shuffle(grid);
270     }
271     
272 }

274 /**
275  * Changes the box implementation when it is revealed  the number of mines adjacent to that spot will appear. If they have flagged every
276  * mine, they win
277  *
278  * @pre The board has been created and filled with mines, the user is playing
279  * @post if the user clicks a mine, they lose. If they click anything other than a mine,
280  * @return none
281  */
282 function mousePressed() {
283     /** Gets coordinate of the click input */
284     let x = floor(mouseX/width);
285     let y = floor(mouseY/width);
286     if (mouseButton === LEFT) {
287         reveal(x,y);
288     }
289     else if (mouseButton === RIGHT) {
290         flag(x,y);
291     }
292     if(isIntense()){
293         if(Math.floor(Math.random() * 20) > 0 ){
294             spinIt();
295             shakeIt();
296         }
297     }
298     
299 }
