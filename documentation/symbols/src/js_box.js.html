 12  /**
 13   * A class to represent boxes in the grid
 14   * @class
 15   *
 16 
 17   *
 18   * @property {number} x the box's x coordinate
 19   * @property {number} y the box's y coordinate
 20   * @property {number} w the box's width
 21   * @property {boolean} mine  contains or not a mine
 22   * @property {boolean} revealed  whether or not the box is revealed
 23   * @property {number} count number of mines surrounding this space
 24   * @property {boolean} flagged   contains or not a flag
 25   * @property {boolean} hasBeenRevealed tracks whether the space has been revealed during the session
 26   */
 27 
 28 
 29 /**
 30   * @constructor
 31   * @pre: need valid grid inputs
 32   * @Post: a box object which is available to be shown has been generated
 33   * @return: undefined
 34 */
 35 function Box(x,y,width)
 36 {
 37     this.x = x;
 38     this.y = y;
 39     this.w = width;
 40     this.mine = false;
 41     this.revealed = false;
 42     this.count = 0;
 43     this.flagged = false;
 44     this.hasBeenRevealed = false;
 45 }
 46
